import IconsOptional from "/snippets/icons-optional.mdx";

Verwenden Sie Schritte, um eine Abfolge von Aktionen oder Ereignissen anzuzeigen. Sie können beliebig viele Schritte hinzufügen.

<Steps>
  <Step title="Erster Schritt">
    Dies sind Anweisungen oder Inhalte, die nur für den ersten Schritt relevant sind.
  </Step>
  <Step title="Zweiter Schritt">
    Dies sind Anweisungen oder Inhalte, die nur für den zweiten Schritt relevant sind.
  </Step>
  <Step title="Dritter Schritt">
    Dies sind Anweisungen oder Inhalte, die nur für den dritten Schritt relevant sind.
  </Step>
</Steps>

```mdx Steps example
<Steps>
  <Step title="First Step">
    These are instructions or content that only pertain to the first step.
  </Step>
  <Step title="Second Step">
    These are instructions or content that only pertain to the second step.
  </Step>
  <Step title="Third Step">
    These are instructions or content that only pertain to the third step.
  </Step>
</Steps>
```

<div id="steps-properties">
  ## Eigenschaften von Steps
</div>

<ResponseField name="children" type="ReactElement<StepProps>[]" required>
  Eine Liste von `Step`-Komponenten.
</ResponseField>

<ResponseField name="titleSize" type="string" default="p">
  Die Größe der Step-Titel. Entweder `p`, `h2` oder `h3`.
</ResponseField>

<div id="individual-step-properties">
  ## Eigenschaften eines einzelnen Schritts
</div>

<ResponseField name="children" type="string | ReactNode" required>
  Der Inhalt eines Schritts, entweder als reiner Text oder als Komponenten.
</ResponseField>

<IconsOptional />

<ResponseField name="title" type="string">
  Der Titel ist der primäre Text des Schritts und wird neben dem Indikator angezeigt.
</ResponseField>

<ResponseField name="stepNumber" type="number">
  Die Nummer des Schritts.
</ResponseField>

<ResponseField name="titleSize" type="string" default="p">
  Die Größe der Schritttitel. Entweder `p`, `h2` oder `h3`.
</ResponseField>
