export const SearchBar = () => {
  const focusSearch = () => {
    const searchInput = document.querySelector("#search-input, [data-search-input], .search-input");
    if (searchInput) {
      searchInput.focus();
    } else {
      document.dispatchEvent(new KeyboardEvent("keydown", { key: "k", metaKey: true }));
    }
  };
  return (
  <div
    role="button"
    tabIndex={0}
    onClick={focusSearch}
    onKeyDown={(e) => e.key === "Enter" && focusSearch()}
    className="w-full max-w-[480px] mx-auto flex items-center gap-2 px-4 py-2.5 rounded-full bg-[var(--background-subtle)] dark:bg-[var(--background-subtle)] border border-[var(--border-subtle)] dark:border-[var(--border-subtle)] text-[var(--foreground-muted)] dark:text-[var(--foreground-muted)] cursor-pointer hover:border-[var(--foreground-muted)] transition-colors"
  >
    <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className="shrink-0">
      <circle cx="11" cy="11" r="8" />
      <path d="m21 21-4.35-4.35" />
    </svg>
    <span className="text-base">Search or ask...</span>
  </div>
  );
};
