使用步骤来展示一系列按顺序进行的操作或事件。你可以根据需要添加任意数量的步骤。

<Steps>
  <Step title="第一步">
    这些是仅与第一步相关的说明或内容。
  </Step>

  <Step title="第二步">
    这些是仅与第二步相关的说明或内容。
  </Step>

  <Step title="第三步">
    这些是仅与第三步相关的说明或内容。
  </Step>
</Steps>

```mdx Steps 示例
<Steps>
  <Step title="第一步">
    以下说明或内容仅适用于第一步。
  </Step>
  <Step title="第二步">
    以下说明或内容仅适用于第二步。
  </Step>
  <Step title="第三步">
    以下说明或内容仅适用于第三步。
  </Step>
</Steps>
```

<div id="steps-properties">
  ## Steps 属性
</div>

<ResponseField name="children" type="ReactElement<StepProps>[]" required>
  `Step` 组件的列表。
</ResponseField>

<ResponseField name="titleSize" type="string" default="p">
  步骤标题的尺寸。可选值：`p`、`h2`、`h3`。
</ResponseField>

<div id="individual-step-properties">
  ## 单个步骤属性
</div>

<ResponseField name="children" type="string | ReactNode" required>
  步骤内容，可以是纯文本或组件。
</ResponseField>

<IconsOptional />

<ResponseField name="title" type="string">
  标题是该步骤的主要文本，显示在指示器旁。
</ResponseField>

<ResponseField name="stepNumber" type="number">
  该步骤的序号。
</ResponseField>

<ResponseField name="titleSize" type="string" default="p">
  步骤标题的大小。可选值为 `p`、`h2`、`h3`。
</ResponseField>
